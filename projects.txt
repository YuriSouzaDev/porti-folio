# Projeto de bliblioteca para melhor entendimento de metodos do javascript

# Array
# Métodos de acesso: concat, join, slice, indexOf, lastIndexOf, includes.
# Métodos iterativos: forEach, every, some, filter, map, reduce, reduceRight, find, findIndex.
# Métodos de mutação: push, pop, shift, unshift, splice, sort, reverse, fill, copyWithin.
# String
# Métodos de acesso: charAt, charCodeAt, codePointAt, indexOf, lastIndexOf, startsWith, endsWith, includes, slice, substring, substr.
# Métodos de mutação e transformação: concat, repeat, replace, replaceAll, toLowerCase, toUpperCase, trim, trimStart, trimEnd, split, padStart, padEnd.
# Object
# Métodos estáticos: assign, create, defineProperties, defineProperty, entries, freeze, getOwnPropertyDescriptor, getOwnPropertyDescriptors, getOwnPropertyNames, getOwnPropertySymbols, getPrototypeOf, is, isExtensible, isFrozen, isSealed, keys, preventExtensions, seal, setPrototypeOf, values.
# Math
# Funções gerais: abs, ceil, floor, max, min, random, round, sign, trunc.
# Funções de exponenciação e logaritmos: exp, expm1, log, log1p, log10, log2, pow, sqrt.
# Funções trigonométricas: acos, acosh, asin, asinh, atan, atan2, atanh, cos, cosh, sin, sinh, tan, tanh.
# Date
# Métodos de criação: now, parse, UTC.
# Métodos de instância (getters): getDate, getDay, getFullYear, getHours, getMilliseconds, getMinutes, getMonth, getSeconds, getTime, getTimezoneOffset, getUTCDate, getUTCDay, getUTCFullYear, getUTCHours, getUTCMilliseconds, getUTCMinutes, getUTCMonth, getUTCSeconds.
# Métodos de instância (setters): setDate, setFullYear, setHours, setMilliseconds, setMinutes, setMonth, setSeconds, setTime, setUTCDate, setUTCFullYear, setUTCHours, setUTCMilliseconds, setUTCMinutes, setUTCMonth, setUTCSeconds.
